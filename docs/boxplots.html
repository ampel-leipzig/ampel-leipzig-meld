library("targets")
library("beeswarm")
library("viridisLite")
tar_load(imp_data)
tar_load(zlog_data)

.bp <- function(x, formula, col = cividis(ncol(x)), main = NULL,
                outline = FALSE) {
    boxplot(
        formula, data = x, col = gsub("..$", "40", col),
        pch = NA, outline = outline,
        main = main
    )
    beeswarm(formula, data = x, col = col, add = TRUE, pch = 19L)
}
