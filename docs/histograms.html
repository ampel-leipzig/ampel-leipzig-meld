.qqnorm <- function(y, ..., yname = deparse(substitute(y)),
    main = paste0("Normal Q-Q Plot (", yname, ")"),
    ylab = paste0("Sample Quantiles (", yname, ")"),
    col = palette.colors(2L)[2L]) {
    qqnorm(y, ..., main = main, ylab = ylab, col = col, pch = 19L)
    qqline(y, ..., col = col)
}
